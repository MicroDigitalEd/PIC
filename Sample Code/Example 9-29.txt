#include <p18f4580.h>
void T0Delay(void);
#define mybit PORTBbits.RB4
void main(void)
  {
   TRISBbits.TRISB4=0;
   while(1)
    { 
      mybit^=1;				
      T0Delay();				
    }
  }
void T0Delay()
  {	
    T0CON=0x01;			
    TMR0H=0x85;				
    TMR0L=0xEE;				
    T0CONbits.TMR0ON=1;			
    while(INTCONbits.TMR0IF==0);	
    T0CONbits.TMR0ON=0;			
    INTCONbits.TMR0IF=0;		
  }
