#include <p18f458.h>
#define SW PORTDbits.RD7
#define ENABLE PORTBbits.RB0
#define MTR_1 PORTBbits.RB1
#define MTR_2 PORTBbits.RB2
void main()
  {
  TRISD=0x80;   
  TRISB=0x0;    
  SW = 1;
  ENABLE = 0;
  MTR_1 = 0;
  MTR_2 = 0;
  while(1)
    {
      ENABLE = 1;
      if(SW == 1)
        {
          MTR_1 = 1;
          MTR_2 = 0;
        }
      else
        {
          MTR_1 = 0;
          MTR_2 = 1;
        }
    }
  }
