#include <p18f4580.h>
void T0Delay(void);
void main(void)
  {
    unsigned char x;
    TRISC=0;			
    PORTC=0x55;
    while(1)
      {
        PORTC=~PORTC;		
        for(x=0;x<20;x++)
          T0Delay();
      }
  }
void T0Delay()
  {	
    T0CON=0x0;		
    TMR0H=0xFF;		
    TMR0L=0x06;		
    T0CONbits.TMR0ON=1;			
    while(INTCONbits.TMR0IF==0);	
    T0CONbits.TMR0ON=0;			
    INTCONbits.TMR0IF=0;			
  }
