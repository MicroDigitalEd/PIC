#include <p18F458.h>
#define LED1 PORTCbits.RC4
#define LED2 PORTCbits.RC5
void chk_isr(void);
void RBINT_ISR(void);
#pragma code My_HiPrio_Int =0x0008 
void My_HiPrio_Int (void)
{
	_asm
		GOTO chk_isr
	_endasm
}
#pragma code
#pragma interrupt chk_isr	
void chk_isr (void)
{
	if (INTCONbits.RBIF==1)	
	RBINT_ISR( );		
}
void main(void)
  {
   	TRISCbits.TRISC4=0;	
	TRISCbits.TRISC5=0;	
	TRISBbits.TRISB4 = 1;	
	TRISBbits.TRISB5 = 1;	
	INTCONbits.RBIF=0; 	
	INTCONbits.RBIE=1;	
	INTCONbits.GIE=1;	
	while(1);	
  }
void RBINT_ISR(void)
  {
	LED1=PORTBbits.RB4;
	LED2=PORTBbits.RB5;
	INTCONbits.RBIF=0;	
  }
