#include <P18F458.INC>
NUME		EQU	0x15		
QU		EQU	0x20		
RMND_L	EQU	0x22
RMND_M	EQU	0x23
RMND_H	EQU	0x24
MYNUM		EQU	0xFD		
MYDEN		EQU	D'10'		
		ORG 	0H		
		MOVLW MYNUM 	
		MOVWF NUME		
		MOVLW MYDEN		
		CLRF  QU,F		
D_1		INCF  QU,F		
		SUBWF NUME		
		BC   D_1		
		ADDWF NUME		
		DECF QU,F		
		MOVFF NUME,RMND_L 
		MOVFF QU,NUME	
		CLRF  QU		
D_2		INCF  QU,F
		SUBWF NUME		
		BC	D_2		
		ADDWF NUME		
		DECF QU,F
		MOVFF NUME,RMND_M 
		MOVFF QU,RMND_H	
HERE		GOTO HERE		
		END
