#include <P18F4580.h>
#define ldata PORTD		
#define rs PORTBbits.RB0	
#define rw PORTBbits.RB1	
#define en PORTBbits.RB2	
void main()
  { 
    TRISD = 0;			
    TRISB = 0;
    en = 0;           		
    MSDelay(250);
    lcdcmd(0x38);			
    MSDelay(250);
    lcdcmd(0x0E);			
    MSDelay(15);
    lcdcmd(0x01);			
    MSDelay(15);
    lcdcmd(0x06);			
    MSDelay(15);	
    lcdcmd(0x86);			
    MSDelay(15);	
    lcddata('M');			
    MSDelay(15);	
    lcddata('D');  	    	
    MSDelay(15);	
    lcddata('E');  	    	
  }
void lcdcmd(unsigned char value)
  {
    ldata = value;  		
    rs = 0;
    rw = 0;
    en = 1;         		
    MSDelay(1);
    en = 0;
  }
void lcddata(unsigned char value)
  {
    ldata = value;  		
    rs = 1;
    rw = 0;
    en = 1;         		
    MSDelay(1);
    en = 0;
  }
void MSDelay(unsigned int itime)
  {
    unsigned int i, j;
    for(i=0;i<itime;i++)
      for(j=0;j<135;j++);
  }
