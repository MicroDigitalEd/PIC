#include <p18f4580.h>
void T0M8Delay(void);
#define mybit PORTBbits.RB5
void main(void)
  { 
	unsigned char x,y;
    TRISBbits.TRISB5 = 0;
    while(1)
      {
        mybit^=1;			
        for(x=0;x<250;x++)		
          for(y=0;y<35;y++)	
            T0M8Delay();
      }
  }
void T0M8Delay()
  {	
    T0CON=0x45;		
    TMR0L=-1;				
    T0CONbits.TMR0ON=1;		
    while(INTCONbits.TMR0IF==0);	
    T0CONbits.TMR0ON=0;		
    INTCONbits.TMR0IF=0;		
  }
