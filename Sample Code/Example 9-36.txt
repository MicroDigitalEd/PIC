#include <p18f4580.h>
void BinToASCII(unsigned char);
void main()
  {
    unsigned char value;
    TRISAbits.TRISA4=1;		
	TRISB=0; 				
	TRISC=0; 				
	TRISD=0; 				
	TMR0L=0;   
 	T0CON=0x65;	
    while(1)
      {
       do
        {
          T0CONbits.TMR0ON=1;	
          value=TMR0L;
          BinToASCII(value);
        }
    while(INTCONbits.TMR0IF==0);	
    T0CONbits.TMR0ON=0;		
    INTCONbits.TMR0IF=0;		
      }
  }
void BinToASCII(unsigned char value)		
  {
    unsigned char x,d1,d2,d3;
    x=value/10;
    d1=value%10;
    d2=x%10;
    d3=x/10;
    PORTB=0x30 | d1;
    PORTC=0x30 | d2;
    PORTD=0x30 | d3;
  }
