#include <p18f4580.h>
void T1Delay(void);
#define mybit PORTBbits.RB1
void main(void)
  {
	TRISBbits.TRISB1 = 0;
    while(1)
      {
        mybit^=1;		
          T1Delay();
      }
  }
void T1Delay()
  {	
    T1CON=0x0;		
    TMR1H=0xFE;		
    TMR1L=0x0C;		
    T1CONbits.TMR1ON=1;			
    while(PIR1bits.TMR1IF==0);	
    T1CONbits.TMR1ON=0;			
    PIR1bits.TMR1IF=0;			
  }
