#include <p18f4580.h>
void T0Delay(void);
void main(void)
  {
  TRISB=0;				
  while(1)				
    {
      PORTB=0x55;			
      T0Delay();			
      PORTB=0xAA;			
      T0Delay();
    }
  }
void T0Delay()
  {	
    T0CON=0x08;			
    TMR0H=0x35;			
    TMR0L=0x00;			
    T0CONbits.TMR0ON=1;			
    while(INTCONbits.TMR0IF==0);	
    T0CONbits.TMR0ON=0;			
    INTCONbits.TMR0IF=0;	
  }
