#include <P18F458.h>
void MSDelay(unsigned int);
void main(void)
  {
    TRISB = 0;
    TRISC = 0;			
    PORTB = 0x55;
    PORTC = 0xAA;
    while(1)
      {
        PORTB = ~PORTB;
        PORTC = ~PORTC;
        MSDelay(250);
      }
  }
void MSDelay(unsigned int itime)
  {
    unsigned int i; 
    unsigned char j;
    for(i=0;i<itime;i++)
      for(j=0;j<165;j++);
  }
